def build(env)
  begin
    builder = Environment::environment_builder_factory(env, "#{options[:git_path]}", "#{options[:workspace]}", "#{options[:config_file]}", "#{options[:tmp_folder]}")
    builder.build
  rescue Exception => e
    $logger.error e.message
  end
end
