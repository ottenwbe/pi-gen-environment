def start
  $logger.info '[Start] pi-image in local vagrant environment'
  Dir.chdir(VAGRANT_PATH) do
    system 'vagrant destroy -f' # cleanup old environment
    system 'vagrant up --provider=virtualbox --no-provision'
  end
end
