def initialize(config_path = '', tmp_directory = '', modifier_gem_path = '')
  self.config_path = config_path
  self.tmp_directory = tmp_directory
  self.modifier_gem_path = modifier_gem_path
  $logger.debug "Local Config at '#{@config_path}' with tmp dir '#{@tmp_directory}' and modifier_gem_path = '#{@modifier_gem_path}'"
end
