def self.make_lite(working_dir)
  Dir.chdir working_dir do |d|
    system 'ls stage4'
    system "touch #{d}/stage3/SKIP #{d}/stage4/SKIP #{d}/stage5/SKIP"
    system "rm #{d}/stage4/EXPORT* #{d}/stage5/EXPORT*"
  end
end
